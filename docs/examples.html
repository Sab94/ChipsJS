Load data with Material design

            let demoTwo = $(".demo2").chip({
  separetor : [' ', ',', '+', '-'],
  md: true,
  wrapperStyle: 'margin: 10px;',
});

demoTwo.load(['tags', 'contacts', 'emails']);

Type email and activeOnBlur

            let demoThree = $(".demo3").chip({
  separetor : [' ', ',', '+', '-'],
  activeOnBlur: true,
  wrapperStyle: 'margin: 10px;',
  type: 'email'
});

Set target

            let demoFour = $(".demo4").chip({
  separetor : [' ', ',', '+', '-'],
  activeOnBlur: true,
  md: true,
  target: '#target',
  wrapperStyle: 'margin: 10px;',
  containerStyle: 'display: flex;flex-wrap: wrap;justify-content: center;'
});
